def initialize(path: nil, port: nil, log: nil, args: nil)
  args ||= []
  unless args.any? { |arg| arg.include?('--connect-existing') || arg.include?('--websocket-port') }
    args << '--websocket-port'
    args << '0'
  end
  super
end
